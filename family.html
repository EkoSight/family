    <input type="text" id="familyName" placeholder="Enter family name">

    <button onclick="addFamily()">Add Family</button>

    <div id="familyTree"></div>


    <script>

        function addFamily() {

            var familyName = document.getElementById('familyName').value;

            if (familyName.trim() === '') {

                alert('Please enter a family name.');

                return;

            }

            var familyTree = document.getElementById('familyTree');

            var newBranch = document.createElement('div');

            newBranch.className = 'branch';

            var newMember = document.createElement('div');

            newMember.className = 'member';

            newMember.textContent = familyName;

            newBranch.appendChild(newMember);

            var addBranchButton = document.createElement('button');

            addBranchButton.textContent = 'Add Branch';

            addBranchButton.onclick = function() {

                addBranch(newBranch);

            };

            newBranch.appendChild(addBranchButton);

            familyTree.appendChild(newBranch);

        }


        function addBranch(parentBranch) {

            var branchName = prompt('Enter branch name:');

            if (branchName.trim() === '') {

                alert('Branch name cannot be empty.');

                return;

            }

            var newBranch = document.createElement('div');

            newBranch.className = 'branch';

            var newMember = document.createElement('div');

            newMember.className = 'member';

            newMember.textContent = branchName;

            newBranch.appendChild(newMember);

            var addBranchButton = document.createElement('button');

            addBranchButton.textContent = 'Add Branch';

            addBranchButton.onclick = function() {

                addBranch(newBranch);

            };

            newBranch.appendChild(addBranchButton);

            parentBranch.appendChild(newBranch);

        }

    </script>
